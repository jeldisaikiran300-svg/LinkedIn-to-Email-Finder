import os
import pandas as pd
import logging
from dotenv import load_dotenv

from apollo_api import ApolloClient
from linkedin_scraper import LinkedInScraper

# Setup logging
logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
logger = logging.getLogger(__name__)

def main():
    # Load environment variables
    load_dotenv()
    
    apollo_api_key = os.getenv("APOLLO_API_KEY")
    li_at_cookie = os.getenv("LINKEDIN_LI_AT_COOKIE")

    if not apollo_api_key or apollo_api_key == "your_apollo_api_key_here":
        logger.error("Please set your APOLLO_API_KEY in the .env file.")
        return

    # Initialize Clients
    apollo_client = ApolloClient(api_key=apollo_api_key)
    
    # Optional: Initialize Scraper if you want to scrape LinkedIn first
    # scraper = LinkedInScraper(li_at_cookie=li_at_cookie)
    
    # Load input data
    input_file = "input.csv"
    if not os.path.exists(input_file):
        logger.error(f"{input_file} not found. Please create it with a 'linkedin_url' column.")
        return

    df = pd.read_csv(input_file)
    if "linkedin_url" not in df.columns:
        logger.error("Input CSV must contain a 'linkedin_url' column.")
        return

    results = []

    for index, row in df.iterrows():
        linkedin_url = row["linkedin_url"]
        logger.info(f"Processing ({index+1}/{len(df)}): {linkedin_url}")
        
        # NOTE: Apollo's Match API is highly effective using just the LinkedIn URL.
        # So we can bypass the manual scrape and query Apollo directly.
        # If you strictly want to scrape first to extract info before using Apollo:
        # 
        # scraped_data = scraper.scrape_profile(linkedin_url)
        # print("Scraped Data:", scraped_data)

        # Find email via Apollo
        apollo_data = apollo_client.match_person_by_linkedin(linkedin_url)
        
        result_row = {
            "linkedin_url": linkedin_url,
            "apollo_name": apollo_data.get("name") if apollo_data else None,
            "apollo_email": apollo_data.get("email") if apollo_data else None,
            "apollo_title": apollo_data.get("title") if apollo_data else None,
            "apollo_company": apollo_data.get("company") if apollo_data else None,
        }
        
        results.append(result_row)

    # Clean up scraper if initialized
    # scraper.close()

    # Save results to output CSV
    output_df = pd.DataFrame(results)
    output_file = "output.csv"
    output_df.to_csv(output_file, index=False)
    logger.info(f"Process complete. Results saved to {output_file}")

if __name__ == "__main__":
    main()
